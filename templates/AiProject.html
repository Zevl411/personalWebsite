#                         algorithm described in the Bayes Network chapter 
#                         of the Russel/Norvig textbook.
# ----------------------------------------------------------------------------

# Computes the probability queried, that var has the value val
# using e, variables we already know are True or False, and 
# bn, the network which holds the probability tables 
def pr(var, val, e, bn):
    parents = bn[var][0]
    if len(parents) == 0:
        truePr = bn[var][1][None]
    else:
        parentVals = [e[parent] for parent in parents]
        truePr = bn[var][1][tuple(parentVals)]
    if val==True: return truePr
    else: return 1.0-truePr

# This function normalizes the probabilities so they add to 1
def normalize(QX):
    total = 0.0
    for val in QX.values():
        total += val
    for key in QX.keys():
        QX[key] /= total
    return QX

# Performs the enumeration ask algorithm to find the probability
# P(X|e): P(J|M)
# Input:
#   X - JohnCalls
#   e - MaryCalls is True
#   bn - The Bayesian Network
#   varss - Bottum Up list of the bn
def enumeration_ask(X, e, bn,varss):
    QX = {}
    for xi in [True,False]:
        e[X] = xi
        QX[xi] = enumerate_all(varss,e,bn)
        del e[X]
    return normalize(QX)

# Performs the recursion by following the call tree
def enumerate_all(varss, e,bn):
    if len(varss) == 0: return 1.0
    Y = varss.pop()
    if Y in e:
        val = pr(Y,e[Y],e,bn) * enumerate_all(varss,e,bn)
        varss.append(Y)
        return val
    else:
        total = 0
        e[Y] = True
        total += pr(Y,True,e,bn) * enumerate_all(varss,e,bn)
        e[Y] = False
        total += pr(Y,False,e,bn) * enumerate_all(varss,e,bn)
        del e[Y]
        varss.append(Y)
        return total

import networkx as nx

# Node class that has NODE objects hold their 
# id(name), parent nodes, and probability table.
# The bayesian network dictionary is created when a new node is added.
class NODE:
  def __init__(self, id, parents, values, bn):
    self.id = id
    bn[id] = [parents,values]
    self.parents = parents

# DAG class that represents the Directed Acyclical Graph. 
# The graph adds edges from the node's parents to the node.
class DAG:
  def __init__(self):
    self.graph = nx.DiGraph()
    self.n_nodes = 0

  def add_node(self, node):
    for parent in node.parents:
      self.graph.add_edge(parent, node.id)
    
    self.n_nodes += 1
    
  def print_graph(self):
    nx.draw_random(self.graph, arrows=True, with_labels=True)
